#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>
#include <errno.h>

int main(int argc, char *argv[]) {
    if (argc != 3) {
        printf("Использование: %s <исходный_файл> <результирующий_файл>\n", argv[0]);
        return 1;
    }

    // Открываем исходный файл с использованием системных вызовов
    int source_fd = open(argv[1], O_RDONLY);
    if (source_fd == -1) {
        perror("Ошибка открытия исходного файла");
        return 1;
    }

    // Открываем результирующий файл с использованием системных вызовов
    int result_fd = open(argv[2], O_WRONLY | O_CREAT | O_TRUNC, 0666); // 0666 - разрешения на созданный файл
    if (result_fd == -1) {
        perror("Ошибка открытия результирующего файла");
        close(source_fd);
        return 1;
    }

    char buffer[1024];
    ssize_t bytes_read, bytes_written;

    while ((bytes_read = read(source_fd, buffer, sizeof(buffer))) > 0) {
        bytes_written = write(result_fd, buffer, bytes_read);
        if (bytes_written == -1) {
            perror("Ошибка записи в результирующий файл");
            close(source_fd);
            close(result_fd);
            return 1;
        }
    }

    if (bytes_read == -1) {
        perror("Ошибка чтения исходного файла");
        close(source_fd);
        close(result_fd);
        return 1;
    }

    close(source_fd);
    close(result_fd);

    printf("Файл успешно скопирован.\n");
    return 0;
}
//./копирование_файла <исходный_файл> <результирующий_файл>
